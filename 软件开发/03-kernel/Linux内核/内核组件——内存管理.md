# 内存映射mmap使用篇

## mmap基本原理

mmap() 必须以 PAGE_SIZE 为单位进行映射，实际上，内存只能以页为单位进行映射。

学习目的：mmap的作用是把设备内存直接映射到用户地址空间，这样用户访问这片地址空间就等同于直接访问设备内存了。而DMA算是和mmap功能是搭档关系，用户空间程序把数据存到mmap区后，并且达到一定触发量，驱动这边的DMA就会启动，此时CPU就可以腾出手来做其他事，DMA启动后就会把数据搬运到设备内存。

```cpp
/**
 * @brief XXX
 * @param addr 用户进程空间起始地址,为了移植性,一般给NULL,由内核来选一个合适地址.
 * @param len  映射到用户进程地址空间的大小.
 * @param prot 映射区域的读写属性等.
 * @param ---> PROT_READ (可读)
 * @param ---> PROT_WRITE (可写)
 * @param ---> PROT_EXEC (可执行)
 * @param ---> PROT_NONE (不可访问)
 * @param flags XXX
 * @param --->  MAP_SHARED 创建一个共享映射区域,
 * 多个进程可以通过共享映射方式来映射一个文件,
 * 这样其他进程也可以看到映射内容的改变,
 * 修改后的内容会同步到磁盘文件中.
 * @param --->  MAP_PRIVATE 创建一个私有的写时复制的映射,
 * 多个进程可以通过私有映射的方式来映射一个文件,
 * 这样其他进程不会看到映射内容的改变,
 * 修改后的内容也不会同步到磁盘文件中.
 * @param --->  MAP_ANONYMOUS 创建一个匿名映射，即没有关联到文件的映射。
 * @param --->  MAP_FIXED 使用参数 addr 创建映射,
 * 如果内核中无法映射指定的地址 addr , 那 mmap 会返回失败,
 * 参数 addr 要求按页对齐，
 * 如果指定区域与已有VMA区域重叠,则 do_munmap() 这块区域,然后重新映射新内容.
 * @param --->  MAP_POPULATE 对于文件映射来说，会提前预读文件内容到映射区,
 * 该特性只支持私有映射.
 * @param offset 在文件映射时，表示文件的偏移量.
 * @return NULL 表示失败.
 */
void *mmap(void *addr, size_t len, int prot, int flags, int fd, off_t offset);
```

## 驱动中mmap实现

下面提供了实例代码，可直接在x86虚拟Linux下编译和测试。

驱动程序源代码：mmap_drv.c
```cpp
#include <linux/kernel.h>
#include <linux/module.h>
#include <linux/cdev.h>
#include <linux/fs.h>
#include <linux/pagemap.h>
#include <linux/slab.h>
#include <asm/io.h>
#include <asm/pgtable.h>
#include <asm/cacheflush.h>
#include <linux/uaccess.h>

#define MAX_DEVICES (5)
#define MAJORBASE 0
#define MINORBASE 0

MODULE_LICENSE("GPL");
static dev_t device_number;
static bool dynamic = true;
static struct class *my_class;
#define DEVICE_MEM_SIZE 4096
//static char simulate_device_memory[DEVICE_MEM_SIZE];
static char *simulate_device_memory = NULL;
static struct my_cdev {
	//字符设备实例作为成员包含在自定义设备结构体里
	struct cdev cdev;
	//other element
}my_cdev[MAX_DEVICES];

static int mydevice_open(struct inode *inode, struct file *file)
{
	struct my_cdev *dev = container_of(inode->i_cdev, struct my_cdev, cdev);
	file->private_data = dev;

	pr_info("major:%d, minor:%d, %s\n", 
			MAJOR(dev->cdev.dev), MINOR(dev->cdev.dev), __func__);
	return 0;
}

static int mydevice_release(struct inode *inode, struct file *file)
{
	struct my_cdev *dev = container_of(inode->i_cdev, struct my_cdev, cdev);
	pr_info("major:%d, minor:%d, %s\n", 
			MAJOR(dev->cdev.dev), MINOR(dev->cdev.dev), __func__);

	printk("%s:simulate_device_memory = %s\n", __func__, simulate_device_memory);
	return 0;
}

ssize_t mydevice_read(struct file *file, char __user *user_buffer,
		size_t count, loff_t *offset)
{
	struct my_cdev *dev = (struct my_cdev *)file->private_data;
	pr_info("major:%d, minor:%d, %s\n", 
			MAJOR(dev->cdev.dev), MINOR(dev->cdev.dev), __func__);

	printk("%s:simulate_device_memory = %s\n", __func__, simulate_device_memory);
	return 0;
}

ssize_t mydevice_write(struct file *file, const char __user *user_buffer,
		size_t count, loff_t *offset)
{
	struct my_cdev *dev = (struct my_cdev *)file->private_data;
	pr_info("major:%d, minor:%d, %s\n", 
			MAJOR(dev->cdev.dev), MINOR(dev->cdev.dev), __func__);
	return count;
}

static int mydevice_mmap(struct file *filp, struct vm_area_struct *vma)
{
	unsigned long addr;
	unsigned long size;

	addr = vma->vm_pgoff + 
		(virt_to_phys(simulate_device_memory) >> PAGE_SHIFT);
	size = vma->vm_end - (vma->vm_start + (vma->vm_pgoff << PAGE_SHIFT));
	printk(KERN_INFO"mmap_size from userspace: 0x%lx\n", size);

	if ((vma->vm_pgoff << PAGE_SHIFT) > size)
		return -ENXIO;

	vma->vm_page_prot = pgprot_noncached(vma->vm_page_prot);
	//if (io_remap_pfn_range(vma,//虚拟内存区域，即设备地址将要映射到这里  
	if (remap_pfn_range(vma, 	//虚拟内存区域，即设备地址将要映射到这里  
				vma->vm_start,       //虚拟空间的起始地址  
				addr,                //与物理内存对应的页帧号，物理地址右移12位
				size,                //映射区域大小，一般是页大小的整数倍   
				vma->vm_page_prot))  //保护属性  
	{
		printk(KERN_ERR "io_remap_pfn_range failure (max size:0x%lx)\n", size);
		return -EAGAIN;
	}
	return 0;
}

struct file_operations fops = {
	.owner = THIS_MODULE,
	.open = mydevice_open,
	.release = mydevice_release,
	.read = mydevice_read,
	.write = mydevice_write,
	.mmap = mydevice_mmap,
};

static int mynull_device_init(void)
{
	int retval;
	int i = 0;

	simulate_device_memory = kmalloc(DEVICE_MEM_SIZE, GFP_KERNEL);
	if (!simulate_device_memory)
		return -1;

	pr_info("%s: In init\n", __func__);
	if (dynamic)
	{
		retval = alloc_chrdev_region(&device_number, 0, MAX_DEVICES, "embedded");
	}
	else
	{
		device_number = MKDEV(MAJORBASE, MINORBASE);
		retval = register_chrdev_region(device_number, MAX_DEVICES, "embedded");
	}
	if (!retval)
	{
		int major = MAJOR(device_number);
		dev_t my_device;
		my_class = class_create(THIS_MODULE, "my_driver_class");
		for (i = 0; i < MAX_DEVICES; i++)
		{
			my_device = MKDEV(major, MINORBASE + i);
			cdev_init(&my_cdev[i].cdev, &fops);
			retval = cdev_add(&my_cdev[i].cdev, my_device, 1);
			if (retval)
			{
				pr_info("%s: Failed in adding cdev to subsystem "
						"retval:%d\n",
						__func__, retval);
			}
			else
			{
				device_create(my_class, NULL, my_device, NULL, "my_null%d", i);
			}
		}
	}
	else
		pr_err("%s: Failed in allocating device number "
				"Error:%d\n",
				__func__, retval);
	return retval;
}

static void mynull_device_exit(void)
{
	int i = 0;
	int major = MAJOR(device_number);
	dev_t my_device;
	for (i = 0; i < MAX_DEVICES; i++)
	{
		my_device = MKDEV(major, MINORBASE + i);
		device_destroy(my_class, my_device);
		cdev_del(&my_cdev[i].cdev);
	}
	class_destroy(my_class);
	unregister_chrdev_region(device_number, MAX_DEVICES);
	pr_info("%s: In exit\n", __func__);
}

module_init(mynull_device_init);
module_exit(mynull_device_exit);
```

测试例程源代码：mmap_app.c
```cpp
#include <unistd.h>
#include <fcntl.h>
#include <sys/types.h>
#include <sys/stat.h>
#include <sys/mman.h>
#include <sys/ioctl.h>

#include <stdio.h>
#include <stdlib.h>
#include <stdint.h>
#include <stdbool.h>
#include <string.h>
#include <errno.h>

#define MMAP_BYTES (4096)

int main(int argc, char *argv[])
{
	const char *devpath = "/dev/my_null0";
	int devfd = open(devpath, O_RDWR);
	if (devfd < 0)
	{
		fprintf(stderr,"open %s error,errno:%d:%s\n", devpath, errno, strerror(errno));
		return 1;
	}

	char *mapdatabuf = (char *)mmap(NULL, MMAP_BYTES, 
			PROT_READ | PROT_WRITE, MAP_SHARED, devfd, 0);
	if ((mapdatabuf == (void *)-1) || (mapdatabuf == NULL))
	{
		fprintf(stderr,"mmap error,errno:%d:%s\n", errno, strerror(errno));
		return 2;
	}

	//memset(mapdatabuf, 0, MMAP_BYTES);
	strcpy(mapdatabuf, "2333366666\0");
	printf("mapdatabuf = %s\n", mapdatabuf);

	read(devfd, NULL, 0);

	munmap(mapdatabuf, MMAP_BYTES);
	close(devfd);
	return 0;
}
```

## 私有匿名映射
当参数 fd=-1 且 flags=MAP_ANONYMOUS | MAP_PRIVATE 时，就是私有匿名映射。该映射最常见用途是在 glibc 分配大块内存时，当需要分配的内存大于 MMAP_THREASHOLD (128KB) 时，glibc 会默认使用 mmap 代替 brk 来分配内存。

## 共享匿名映射
当参数 fd=-1 且 flags=MAP_ANONYMOUS | MAP_SHARED 时，就是共享匿名映射。主要用于父子进程之间共享数据和通信。

创建共享匿名映射有两种方式：
- 1）fd=-1 且 flags=MAP_ANONYMOUS | MAP_SHARED 时，最终会调用 shmem_zero_setup() 来打开 /dev/zero .
- 2）直接打开 /dev/zero ，然后使用文件描述符fd来创建mmap.
