# Linux内核双向循环链表


```cpp
#ifndef C_API_LIST_H
#define C_API_LIST_H

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <errno.h>

#ifdef __cplusplus
extern "C" {
#endif

#undef offsetof
#ifdef __compiler_offsetof
#define offsetof(TYPE,MEMBER) __compiler_offsetof(TYPE,MEMBER)
#else
#define offsetof(TYPE, MEMBER) ((size_t) &((TYPE *)0)->MEMBER)
#endif

/**
 * container_of - cast a member of a structure out to the containing structure
 * @ptr:  the pointer to the member.
 * @type:  the type of the container struct this is embedded in.
 * @member:  the name of the member within the struct.
 *
 */
#define container_of(ptr, type, member) (type *)((char *)(ptr) - offsetof(type, member))

struct list_head {
  struct list_head *next, *prev;
};

/**
 * list_for_each  -  iterate over a list
 * @pos:  the &struct list_head to use as a loop cursor.
 * @head:  the head for your list.
 */
#define list_for_each(pos, head) \
  for (pos = (head)->next; pos != (head); pos = pos->next)

/**
 * list_for_each_prev  -  iterate over a list backwards
 * @pos:  the &struct list_head to use as a loop cursor.
 * @head:  the head for your list.
 */
#define list_for_each_prev(pos, head) \
  for (pos = (head)->prev; pos != (head); pos = pos->prev)

//List initialization
static inline void INIT_LIST_HEAD(struct list_head *list)
{
  list->next = list;
  list->prev = list;
}

static inline void __list_add(struct list_head *new,
            struct list_head *prev,
            struct list_head *next)
{
  next->prev = new;
  new->next = next;
  new->prev = prev;
  prev->next = new;
}

//Insert node (head insert method)
static inline void list_add(struct list_head *new, struct list_head *head)
{
  __list_add(new, head, head->next);
}

//Insert node (tail insert method)
static inline void list_add_tail(struct list_head *new, 
    struct list_head *head)
{
  __list_add(new, head->prev, head);
}

static inline void __list_del(struct list_head * prev, 
    struct list_head * next)
{
  next->prev = prev;
  prev->next = next;
}

static inline void __list_del_entry(struct list_head *entry)
{
  __list_del(entry->prev, entry->next);
}

static inline void list_del(struct list_head *entry)
{
  __list_del(entry->prev, entry->next);
  entry->next = entry;
  entry->prev = entry;
}

/**
 * list_del_init - deletes entry from list and reinitialize it.
 * @entry: the element to delete from the list.
 */
static inline void list_del_init(struct list_head *entry)
{
  __list_del_entry(entry);
  INIT_LIST_HEAD(entry);
}

//Take the old node out and insert the new node in the old position.
static inline void list_replace(struct list_head *old,
        struct list_head *new)
{
  new->next = old->next;
  new->next->prev = new;
  new->prev = old->prev;
  new->prev->next = new;
}

static inline void list_replace_init(struct list_head *old,
          struct list_head *new)
{
  list_replace(old, new);
  INIT_LIST_HEAD(old);
}

/**
 * list_move - delete from one list and add as another's head
 * @list: the entry to move
 * @head: the head that will precede our entry
 */
static inline void list_move(struct list_head *list, struct list_head *head)
{
  __list_del_entry(list);
  list_add(list, head);
}

/**
 * list_move_tail - delete from one list and add as another's tail
 * @list: the entry to move
 * @head: the head that will follow our entry
 */
static inline void list_move_tail(struct list_head *list,
          struct list_head *head)
{
  __list_del_entry(list);
  list_add_tail(list, head);
}

/**
 * list_is_last - tests whether @list is the last entry in list @head
 * @list: the entry to test
 * @head: the head of the list
 */
static inline int list_is_last(const struct list_head *list,
        const struct list_head *head)
{
  return list->next == head;
}

/**
 * list_empty - tests whether a list is empty
 * @head: the list to test.
 */
static inline int list_empty(const struct list_head *head)
{
  return head->next == head;
}

#ifdef __cplusplus
}
#endif // extern "C"

#endif // C_API_LIST_H
```

Simple student example：
```cpp
#include "list.h"

//user demo
struct student {
    char name[32];
    int  age;
    struct list_head entry;
};

static struct list_head students;

int main(int argc, char const *argv[])
{
    INIT_LIST_HEAD(&students);

    struct student *stu = NULL;

    for (int i = 0; i < 8; i++)
    {
        stu = (struct student *)malloc(sizeof(struct student));
        if (!stu)
        {
            printf("malloc stu error (%s)\n", strerror(errno));
            return -1;
        }

        snprintf(stu->name, sizeof(stu->name), "name%02d", i);
        list_add_tail(&stu->entry, &students);
    }


    struct list_head *pos = NULL;
    list_for_each(pos, &students)
    {
        stu = container_of(pos, struct student, entry);
        printf("list_for_each:stu->name = %s\n", stu->name);
    }

    return 0;
}
```

# Linux内核kfifo

学习方法：代值法，即代入某个简单的具体值来进行分析.

## 核心数据结构

```cpp
struct __kfifo {
  unsigned int  in; //index of in
  unsigned int  out;//index of out
  unsigned int  mask;
  unsigned int  esize;//element size
  void    *data;
};//一个结构体实例就表示一个kfifo
```

初始化函数（关注函数注释）：
```cpp
/**
 * kfifo_alloc - allocates a new FIFO buffer
 * @fifo:  pointer to the fifo to allocate.
 * @size:  how many elements to allocate.
 * @esize:  size of each element.
 * @return 0 if ok, -ENOMEM if no memory available.
 */
int __kfifo_alloc(struct __kfifo *fifo, unsigned int size,
    size_t esize, gfp_t gfp_mask);
void __kfifo_free(struct __kfifo *fifo);

/**
 * kfifo_init - initialize a fifo buffer
 * @fifo:  pointer to the fifo to initialize.
 * @buffer:  pointer to the data buffer
 * @size:  how many elements to allocate.
 * @esize:  size of each element.
 * @return 0 if ok, -ENOMEM if no memory available.
 */
int __kfifo_init(struct __kfifo *fifo, void *buffer,
    unsigned int size, size_t esize);
```

## kfifo使用例子（动态申请）

-   (1) 包含头文件 `#include <linux/kfifo.h>`
-   (2) 定义一个 struct kfifo 变量；
-   (3) 使用kfifo_alloc申请(**apply**)内存空间；
-   (4) 分别使用 kfifo_in 和 kfifo_out 进行入队和出队操作；
-   (5) 当(be no longer used)不再使用 kfifo 时，使用 kfifo_free 释放请求的内存。

```cpp
#include <linux/kfifo.h>

static int xxx_init(void)
{
  struct kfifo kfifo;
  int ret;
  unsigned char data[8];
  ret = kfifo_alloc(&kfifo, 8, GFP_KERNEL);
  ret = kfifo_in(&kfifo, "ABCD", 4);
  ret = kfifo_in(&kfifo, "EF", 4);

  for(i=0; i<128; i++)
  {
    kfifo_out(&kfifo, buf, 4);
    ret = kfifo_in(&kfifo, "ABCD", 4);
  }

  printk("kfifo_len() = %d\n", kfifo_len(&kfifo));
  printk("fifo->kfifo.in = %d, out = %d\n", 
    kfifo.kfifo.in, kfifo.kfifo.out);
}
```

## 总结

如果在使用kfifo的过程中没有清除kfifo->in和kfifo->out，'in'或者'out'的值会不断增加直到溢出。 当然，这可以人为地避免。 在使用kfifo的过程中，如果kfifo为空，可以清除A和B的值，让它们从0开始。

```cpp
if (kfifo_is_empty(&kfifo) && (kfifo->in || kfifo->out)) {
  kfifo->in = 0;
  kfifo->out = 0;
  smp_wmb();
}
```
