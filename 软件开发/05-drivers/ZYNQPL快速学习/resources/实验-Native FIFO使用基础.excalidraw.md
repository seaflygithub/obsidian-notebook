`timescale 1ns / 1ps

module data_source (
    input logic clk_fast,
    input logic rst_n,
    input logic enable,
    input logic ready,

    output logic [31:0] data_out,
    output logic        valid
);

  logic [31:0] data_cnt;
  logic        handshake;

  assign handshake = valid && ready;

  // 更标准写法（避免ready=0时重复拉高问题）
  always_ff @(posedge clk_fast or negedge rst_n) begin
    if (!rst_n) begin
      valid <= 1'b0;
    end
    else if ((!valid) || ready) begin
      valid <= enable;
    end
  end

  always_ff @(posedge clk_fast or negedge rst_n) begin
    if (!rst_n) begin
      data_cnt <= 32'd0;
      data_out <= 32'd0;
    end else if (handshake) begin
      data_cnt <= data_cnt + 1;
      data_out <= data_cnt;
    end
  end

endmodule ^JTHBv01C

`timescale 1ns / 1ps

module top (
    input logic clk_in,  // 50MHz
    input logic rst_n
);

  //====================================================
  // 1. 时钟（两个不同时钟，制造跨域）
  //====================================================
  logic clk_fast;
  logic clk_slow;
  logic locked;

  clk_wiz_0 u_clk (
      .clk_in1(clk_in),
      .clk_out1(clk_fast),  // 比如100MHz
      .clk_out2(clk_slow),  // 比如25MHz
      .reset(~rst_n),
      .locked(locked)
  );

  logic        rst = ~locked;

  //====================================================
  // 2. data_source（写侧）
  //====================================================
  logic [31:0] s_data;
  logic        s_valid;
  logic        s_ready;

  //====================================================
  // 3. Async FIFO（Native 双时钟）
  //====================================================
  logic [31:0] fifo_dout;
  logic        fifo_full;
  logic        fifo_empty;

  assign s_ready = ~fifo_full;  // FIFO没满才能写

  data_source u_src (
      .clk_fast(clk_fast),
      .rst_n(~rst),
      .enable(1'b1),
      .ready(s_ready),
      .data_out(s_data),
      .valid(s_valid)
  );

  logic        wr_en = s_valid & s_ready;



  // 读侧控制
  logic        rd_en;
  assign rd_en = ~fifo_empty;  // 有数据就一直读

  fifo_generator_0 u_fifo (
      .rst(rst),

      // 写侧（clk_fast）
      .wr_clk(clk_fast),
      .din(s_data),
      .wr_en(wr_en),
      .full(fifo_full),

      // 读侧（clk_slow）
      .rd_clk(clk_slow),
      .rd_en (rd_en),
      .dout  (fifo_dout),
      .empty (fifo_empty)

  );

  //====================================================
  // 4. ILA（同时观察两侧）
  //====================================================
  ila_0 u_ila (
      .clk(clk_fast),  // ILA用快时钟


      .probe0(s_data),    // 写数据
      .probe1(s_valid),
      .probe2(s_ready),
      .probe3(wr_en),
      .probe4(fifo_full),

      .probe5(fifo_dout),  // 读数据（跨域观察）
      .probe6(rd_en),
      .probe7(fifo_empty)

  );

endmodule

上板后你应该看到什么（重点）
正常现象：
写侧（clk_fast）
s_valid：一直1
wr_en：大部分时间1
fifo_full：偶尔1（慢时钟读不过来）

读侧（clk_slow）
rd_en：有数据就读
fifo_empty：偶尔1


八、你可以用 ILA 验证的关键点（非常有价值）
1️⃣ 是否丢数据
fifo_dout 是否连续递增

2️⃣ 是否跨时钟正确
写快读慢 → FIFO 会满
写慢读快 → FIFO 会空

3️⃣ 是否有乱序（绝对不能有）
👉 FIFO必须严格有序


九、一个进阶建议（强烈推荐你试）
把读改慢一点：rd_en = (counter[3] == 1);
👉 你会看到：
FIFO变满
backpressure传回data_source
👉 这是理解系统流控的关键一步 ^19ecE58e
