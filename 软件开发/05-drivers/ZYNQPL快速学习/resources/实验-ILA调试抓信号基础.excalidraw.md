`timescale 1ns / 1ps

module top (
    input  wire clk_in,   // 板子50MHz
    input  wire rst_n     // 按键复位（低有效）
);

//====================================================
// 1. 时钟生成（clk_wiz）
//====================================================
wire clk_fast;
wire locked;

clk_wiz_0 u_clk_wiz (
    .clk_in1(clk_in),
    .clk_out1(clk_fast),
    .reset(~rst_n),
    .locked(locked)
);

// 同步复位（推荐）
wire rst_sync = ~locked;


//====================================================
// 2. data_source实例
//====================================================
wire [31:0] data_out;
wire        valid;
wire        ready;
wire        enable;
wire        handshake;

// 👉 可以先固定，保证系统跑起来
assign enable = 1'b1;
assign ready  = 1'b1;

// handshake在top里再打一份（方便ILA接）
assign handshake = valid & ready;
assign data_clk = clk_fast;

data_source u_data_source (
    .clk_fast(clk_fast),
    .rst_n(~rst_sync),
    .enable(enable),
    .ready(ready),
    .data_out(data_out),
    .valid(valid)
);


//====================================================
// 3. ILA
//====================================================
ila_0 u_ila (
    .clk(clk_fast),

    .probe0(data_out),   // [31:0]
    .probe1(data_clk),   // [0:0]
    .probe2(valid),      // [0:0]
    .probe3(ready),      // [0:0]
    .probe4(enable),     // [0:0]
    .probe5(handshake)   // [0:0]
);

endmodule ^w8UEx3C1

`timescale 1ns / 1ps

module data_source (
    input  logic        clk_fast,
    input  logic        rst_n,
    input  logic        enable,
    input  logic        ready,

    output logic [31:0] data_out,
    output logic        valid
);

logic [31:0] data_cnt;
logic        handshake;

// 抽象出握手信号,方便后续修改扩展
assign handshake = valid && ready;

// valid
always_ff @(posedge clk_fast or negedge rst_n) begin
    if (!rst_n)
        valid <= 1'b0;
    else
        valid <= enable;
end

// data path
always_ff @(posedge clk_fast or negedge rst_n) begin
    if (!rst_n) begin
        data_cnt <= 32'd0;
        data_out <= 32'd0;
    end else if (handshake) begin
        data_cnt <= data_cnt + 1;
        data_out <= data_cnt;
    end
end

endmodule ^VBaMkxjx

二、加 ILA（重点） ^M4Wx1aZG

1. Vivado左上菜单 >> IP Catalog >> 搜索ILA >> 双击进入配置界面；

2. ILA配置界面:
    选择Native，不选AXI；
    Probe数量至少2个(这里设置6个)
    Probe0 —— data_out (32bit)
    Probe1 —— fast_clk
    Probe2 —— valid
    Probe3 —— ready
    Probe4 —— enable
    Probe5 —— handshake

3. 例化并连接：
// clk_fast由于是ila工作时钟,所以不能直接probe它
assign data_clk = clk_fast;
ila_0 u_ila (
    .clk(clk_fast),

    .probe0(data_out),   // [31:0]
    .probe1(data_clk),   // [0:0]
    .probe2(valid),      // [0:0]
    .probe3(ready),      // [0:0]
    .probe4(enable),     // [0:0]
    .probe5(handshake)   // [0:0]
); ^3orUmLcc

当有多个 ila 核的时候, 
它们最终会汇总到 dbg_hub 统一和下载器通信;
而且生成网表时会自动选择其中一个时钟来作为 dbg_hub 工作时钟;
但当选择的时钟没有工作,这时候就会导致ila界面加载不起来;
说白了就是hub选择的其中一个时钟有问题导致hub工作不起来;

于是需要手动指定 dbg_hub 工作时钟;
在约束中添加类似如下内容：
connect_debug_port dbg_hub/clk [get_nets [list design_1_i/clk_wiz_0/inst/clk_out2]] ^OzbIeiPV
