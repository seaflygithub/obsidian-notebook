`timescale 1ns / 1ps

module data_source (
    input  logic        clk_fast,
    input  logic        rst_n,
    input  logic        enable,
    input  logic        ready,

    output logic [31:0] data_out,
    output logic        valid
);

logic [31:0] data_cnt;
logic        handshake;

// 抽象出握手信号,方便后续修改扩展
assign handshake = valid && ready;

// valid
always_ff @(posedge clk_fast or negedge rst_n) begin
    if (!rst_n)
        valid <= 1'b0;
    else
        valid <= enable;
end

// data path
always_ff @(posedge clk_fast or negedge rst_n) begin
    if (!rst_n) begin
        data_cnt <= 32'd0;
        data_out <= 32'd0;
    end else if (handshake) begin
        data_cnt <= data_cnt + 1;
        data_out <= data_cnt;
    end
end

endmodule ^GTEfGn2V

`timescale 1ns / 1ps

module data_source_tb;

//--------------------------------------------------
// 信号定义
//--------------------------------------------------
logic clk_fast;
logic rst_n;
logic enable;
logic ready;

logic [31:0] data_out;
logic        valid;

//--------------------------------------------------
// DUT 实例化
//--------------------------------------------------
data_source dut (
    .clk_fast (clk_fast),
    .rst_n    (rst_n),
    .enable   (enable),
    .ready    (ready),
    .data_out (data_out),
    .valid    (valid)
);

//--------------------------------------------------
// 时钟：100MHz (10ns周期)
//--------------------------------------------------
initial clk_fast = 0;
always #5 clk_fast = ~clk_fast;

//--------------------------------------------------
// 激励
//--------------------------------------------------
initial begin
    // 初始值
    rst_n  = 0;
    enable = 0;
    ready  = 0;

    // 复位一段时间
    #20;
    rst_n = 1;

    // 打开数据源，但下游不ready
    #10;
    enable = 1;
    ready  = 0;

    // 观察：此时 valid=1，但 data_out 不变化
    #50;

    // 下游开始接收
    ready = 1;

    // 观察：data_out 开始递增
    #100;

    // 制造背压（ready=0）
    ready = 0;

    // 观察：data_out 应保持不变
    #50;

    // 再恢复
    ready = 1;

    #100;

    // 关闭数据源
    enable = 0;

    #50;

    $finish;
end

//--------------------------------------------------
// 监视输出（打印波形信息）
//--------------------------------------------------
initial begin
    $display("time\tvalid ready data_out");
    $monitor("%0t\t%b     %b     %0d", 
             $time, valid, ready, data_out);
end

//--------------------------------------------------
// 可选：自动检查（简单断言）
//--------------------------------------------------
logic [31:0] last_data;

always @(posedge clk_fast) begin
    if (!rst_n)
        last_data <= 0;
    else begin
        // 如果没有握手成功，数据必须保持
        if (!(valid && ready)) begin
            if (data_out != last_data) begin
                $error("ERROR: data changed without handshake!");
            end
        end
        last_data <= data_out;
    end
end

endmodule ^eG3yqEYT
